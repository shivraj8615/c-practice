Indicate the order of operation evlauted first.
 1. g = 10/5/2/1 
    
    g = 10/5/2/1               Here 10/5 will be evlauted first;
    g = 2/2/1                  Then 2/2 will be evlauted ;
    g = 1/1                    Now 1/1 will be evaluted ;
    g = 1

 2. b = 3/2+5*4/3

   b = 3/2+5*4/3              Here 3/2 will be evlauted;
   b = 1+5*4/3                Now 5*4 will be evlauted;
   b = 1+20/3                 Then 20/3 will be evlauted;
   b = 1+6                    Now 1+6 will be evlauted;
   b = 7

3. a = b = c = 3+4

   a = b = c = 3+4            Here 3+4 will be evlauted;
   a = b = c = 7              Now the value will be assigned to'c'
   c = 7                      Next the value will be assigned to'b'
   b = 7                      Then value will be assigned to'a'
   a = 7

4. x = 2-3+5*2/8%3

  x = 2-3+5*2/8%3             Here 5*2 will be evlauted;
  x = 2-3+10/8%3              Now 10/8 will be evlauted;
  x = 2-3+1%3                 Next 1%3 will be evlauted;
  x = 2-3+1                   Then 2-3 will be evlauted;
  x = -1+1                    Finally -1+1 will be evlauted;
  x = 0

5. z = 5%3/8*3+4

  z = 5%3/8*3+4             Here 5%3 will be evlauted;
  z = 2/8*3+4               Now 2/8 will be evlauted;
  z = 0*3+4                 Then 0*3 will be evlauted;
  z = 0+4                   Now 0+4 will be evlauted;
  z = 4

   y = z = -3%-8/2+7        Here -3%-8 will be evlauted;
